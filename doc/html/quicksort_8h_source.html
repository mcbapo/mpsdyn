8 // Partition the vector orig, from position l to r (pivot in l)
9 // and return the new position of the pivot (pos)
10 template <class T>
11 void partition(std::vector<T>& orig,int l,int r,int& pos){
12  // pivot is the first
13  T p=orig[l]; // pivot
14  int i=l+1; // in the partitioned part, first larger than p
15  for(int j=l+1;j<=r;j++){
16  if(orig[j]<=p){
17  T aux=orig[j];
18  orig[j]=orig[i];
19  orig[i++]=aux;
20  }
21  }
22  orig[l]=orig[i-1];
23  orig[i-1]=p;
24  pos=i-1;
25 }
